git clone <github repo link>
git status
git add <filename>
git add .   # stage all untracked/modified files
git commit -m "<message>"
git restore <file>   #discard all changes in a file
git restore --staged <file> #back to local/unstage; modifications remain
git push origin main 
git push -u origin main -> git push #set upstream so that we dont have to write origin main everytime
git init    #initialize a repo on github
git remote add origin <link>
git remote -v
git branch
git branch -M <name>    #rename a branch
git checkout -b <newbranchname> #create a new branch
git checkout <branchname>   #switch branch
git branch -d <branchname>  #delete a branch *if(current branch != delete branch)
git diff <branchname>   #to compare branches, commits, files and more with the current b,c,f
git merge <branchname>  #to merge branches
merge through PR GUI of github
git pull origin <branch>    #pull remote update to local

git reset <filename>    #reset the staged file
git reset   #reset all staged files
git reset --hard <commitHash>   #reset staged file from github + editor simultaneously

git reset HEAD~1   #reset last one commit of file
git log     #gives commit log with their hash
git reset <commitHash>  #go back to a particular commit eg. 6->2
